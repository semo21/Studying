## Code
```c++
class Solution {
public:
    string reverseWords(string s) {
        vector<string> words;
        string word;
        for(int i = 0; i < s.size(); ++i){
            if(s[i] != ' '){
                word += s[i];
            }else{
                if(!word.empty()){
                    words.push_back(word);
                    word.clear();
                }
            }
        }

        if(!word.empty()){
            words.push_back(word);
        }

        string result;
        for(int i = words.size()-1 ; i >= 0; --i){
            result += words[i];
            if(i > 0){
                result += ' ';
            }
        }

        return result;
    }
};
```

### Constraints:
- 2 <= nums.length <= 105
- -30 <= nums[i] <= 30
- The input is generated such that answer[i] is guaranteed to fit in a 32-bit integer.
