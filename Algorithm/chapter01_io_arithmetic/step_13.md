```c++
#include <iostream>

int main(){
    std::cout 
    << "|\\_/|" << "\n"
    << "|q p|   /}" << "\n"
    << "( 0 )\"\"\"\\" << "\n"
    << "|\"^\"`    |" << "\n"
    << "||_/=\\\\__|" << std::endl;
    return 0;
}

// 문자열 안에 특수문자(\, ", ' 등)를 그대로 출력하려면, 백슬래시(\)를 이용한 이스케이프 시퀀스를 사용해야 함.
// ex) 문자열 -> 출력값
// ex) "\\" -> \ (백슬래시 하나 출력)
// ex) "\"" -> " (큰 따옴표 출력)
// ex) "\n" ->  (줄바꿈 문자 출력)
```
