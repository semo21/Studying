```c++
// C++
#include <iostream>

using namespace std;

int main() {
    // 문제에서 명시했듯이 10^12까지의 범위를 받아야 하기 때문에 int가 아닌 long타입을 사용한다.
	long A, B, C;

	cin >> A, cin >> B, cin >> C;
	cout << A + B + C << endl;
	return 0;
}
```

```python
# Python
A, B, C = map(int, input().split());
```

```c
// C
#include <stdio.h>

int main(void) {
	long A, B, C;
    // C에서 long 출력 커맨드는 %lld이다.
	scanf_s("%lld %lld %lld", &A, &B, &C);

	printf("%lld", A + B + C);
	return 0;
}
```
