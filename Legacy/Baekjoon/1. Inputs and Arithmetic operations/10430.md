```c++
// C++
#include <iostream>

using namespace std;

int main() {
	int A, B, C;
	cin >> A, cin >> B, cin >> C;
	cout << (A + B) % C << endl
		<< ((A % C) + (B % C)) % C << endl
		<< (A*B) % C << endl
		<< ((A % C) *(B % C)) % C << endl;
	return 0;
}
```

```c
// C
#include <stdio.h>

int main(void) {
	int A, B, C;

	scanf_s("%d %d %d", &A, &B, &C);

	printf("%d\n",(A + B) % C);
	printf("%d\n",((A % C) + (B % C)) % C);
	printf("%d\n",(A * B) % C);
	printf("%d\n",((A % C) * (B % C)) % C);
	return 0;
}
```
