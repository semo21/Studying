```c
// C
#include <stdio.h>

int main(void) {
	char id[51];

    // scanf_s에서 몇글자를 입력받을건지 크기를 지정하지 않으면 메모리 주소 오류 발생함. 따라서 sizeof(id)로 크기 지정함.
    // 백준에선 scanf만 사용되지만 VS2022에선 scanf_s를 사용해야해서 scanf_s로 작성. 앞으로도 그럴듯.
	scanf_s("%s", id, sizeof(id));

	printf("%s??!", id);

	return 0;
}
```
