```c++
// C++
#include <iostream>

using namespace std;

int main() {
	int A, B, C;

	cin >> A, cin >> B, cin >> C;

	if (A == B && B == C) {
		cout << 10000 + (1000 * A) << endl;
	}
	else if ((A == B && B != C) || (A != B) && (B == C) || (A == C && A != B)) {
		if (A == B || A == C) {
			cout << 1000 + (100 * A) << endl;
		}
		else {
			cout << 1000 + (100 * B) << endl;
		}
	}
	else {
		int max = 0;
		int arr[] = { A, B, C };
		for (int i = 0; i < 3; i++) {
			if (arr[i] > max) {
				max = arr[i];
			}
		}
		cout << 100 * max << endl;
	}

	return 0;
}

// else if문에서 A==C && A!=B를 새로운 else if문으로 쓰면 조금 더 보기 좋을 것 같다.
```
