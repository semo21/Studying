```c++
// C++
#include <iostream>

using namespace std;

int main() {
	int A, B, C;

	cin >> A, cin >> B;
	cin >> C;

	int t = (60 * A) + B + C;

	if (t >= 1440) {
		t -= 1440;
	}

	cout << t / 60 << " " << t % 60 << endl;
	return 0;
}
```
