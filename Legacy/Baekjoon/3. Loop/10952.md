```c++
// C++
#include <iostream>

using namespace std;

int main() {

	while (true) {
		int a, b;
		cin >> a, cin >> b;
		if (a == 0 && b == 0) {
			break;
		}
		cout << a + b << endl;

	}
	return 0;
}
```
