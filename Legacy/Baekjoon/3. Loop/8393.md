```c++
// C++
// 1. 등차수열의 합을 구하는 공식을 이용한 방법
#include <iostream>

using namespace std;

int main() {
	int n;

	cin >> n;

    // 반드시 n*(n+1)을 먼저 하여 2로 나눠도 소수가 되지 않게 해야함.
    // 2로 먼저 나눌 경우 소수가 나오는 값이라면 int형이기때문에 데이터 타입으로 인한 오차가 발생함.
	cout << n * (n + 1) / 2 << endl;

	return 0;
}

// 2. 반복문 사용
#include <iostream>

using namespace std;

int main() {
	int n;

	cin >> n;

	int sum = 0;
	for (int i = 0; i < n; i++) {
		sum += i + 1;
	}
	cout << sum << endl;

	return 0;
}
```
