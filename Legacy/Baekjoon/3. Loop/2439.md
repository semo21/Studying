```python
# Python
# 1. C계열과 같은 중첩반복문 형식
t = int(input())

for i in range(t):
    for k in range(1, t-i):
        print(" ", end="")
    for j in range(i+1):
        print("*", end="");
    print();

# 2. 출력문에 식으로 표현
# Python
t = int(input())

for i in range(1,t+1):
    print(" "*(t-i)+"*"*i);
```

```c
// C
#include <stdio.h>

int main(void) {
	int t;

	scanf_s("%d", &t);

	for (int i = 0; i < t; i++) {
		for (int k = 1; k < t - i; k++) {
			printf(" ");
		}

		for (int j = 0; j <= i; j++) {
			printf("*");
		}
		printf("\n");
	}

	return 0;
}
```
