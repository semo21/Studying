```c++
// C++
#include <iostream>

using namespace std;

int main() {
	int t;

	cin >> t;

	for (int i = 0; i < t; i++) {
		int a, b;
		cin >> a, cin >> b;

		cout << "Case #" << i + 1 << ": " << a << " + " << b << " = " << a + b << endl;
	}
	return 0;
}

```

```python
# Python
t = int(input())

for i in range(t):
    a, b = map(int, input().split())
    print(f'Case #{i+1}: {a} + {b} = {a+b}')
```

```c
// C
#include <stdio.h>

int main(void) {
	int t;

	scanf("%d", &t);

	for (int i = 0; i < t; i++) {
		int a, b;
		scanf("%d %d", &a, &b);

		printf("Case #%d: %d + %d = %d\n", i + 1, a, b, a + b);
	}

	return 0;
}
```

