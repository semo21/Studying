// C++
// 1. 배열 선언 후 반복문을 통한 풀이
// 시간이 적게 든다.
// 4ms
#include <iostream>

using namespace std;

int main() {
	int n, x;
	cin >> n, cin >> x;

	int arr[10001] = {};

	for (int i = 0; i < n; i++) {
		cin >> arr[i];
	}

	for (int i = 0; i < n; i++) {
		if (arr[i] < x) {
			cout << arr[i] << " ";
		}
	}
}

// 2. 배열 선언 없이 풀이
// 메모리는 같으나 시간이 압도적으로 많이 걸린다.
// 424ms
#include <iostream>

using namespace std;

int main() {
	int n, x;
	cin >> n, cin >> x;

	int a;
	for (int i = 0; i < n; i++) {
		cin >> a;
		if (a < x)	cout << a << " ";
	}
}

// C
// C++과는 달리 C에선 1번과 2번 방법 모두 같은 메모리와 시간 값이 나왔다.
// 1. 배열 선언 후 반복문을 통한 수 찾기
#include <stdio.h>

int main(void) {
	int n, x;
	scanf_s("%d %d", &n, &x);

	int arr[10001] = {0};
	for (int i = 0; i < n; i++) {
		scanf_s("%d", &arr[i]);
	}

	for (int i = 0; i < n; i++) {
		if (arr[i] < x) {
			printf("%d ", arr[i]);
		}
	}
	return 0;
}

// 2. 배열 선언없이 수 찾기
#include <stdio.h>

int main(void) {
	int n, x;
	scanf_s("%d %d", &n, &x);

	int a;
	for (int i = 0; i < n; i++) {
		scanf_s("%d", &a);
		if (a < x) {
			printf("%d ", a);
		}
	}
	return 0;
}
