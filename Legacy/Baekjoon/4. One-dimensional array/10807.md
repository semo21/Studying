```c++
// C++
#include <iostream>

using namespace std;

int main() {
	int n, v;

	cin >> n;

	int arr[101];

	for (int i = 0; i < n; i++) {
		cin >> arr[i];
	}

	cin >> v;
	int r = 0;
	for (int i = 0; i < n; i++) {
		if (arr[i] == v) {
			r += 1;
		}
	}
	cout << r << endl;

	return 0;
}
```

```c
	int n, v;

	scanf_s("%d", &n);

	int arr[101];

	for (int i = 0; i < n; i++) {
		scanf_s("%d", &arr[i]);
	}

	scanf_s("%d", &v);

	int r = 0;
	for (int i = 0; i < n; i++) {
		if (arr[i] == v) {
			r += 1;
		}
	}
	printf("%d\n", r);

	return 0;
}
```
