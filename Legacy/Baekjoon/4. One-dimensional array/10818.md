// C++
// 1.
// 배열 선언 x
// 메모리 2020KB, 시간 460ms
#include <iostream>

using namespace std;

int main() {
	int n;
	cin >> n;

	int max;
	int min;
	int a;
	for (int i = 0; i < n; i++) {
		cin >> a;
		if (i == 0) {
			min = a;
			max = a;
		}

		if (a > max)	max = a;
		if (a < min)	min = a;
	}

	cout << min << " " << max << endl;
}

// 2. 배열 선언 o
// 메모리 5804KB, 시간 464ms
#include <iostream>

using namespace std;

int main() {
	int n;
	cin >> n;

	int arr[1000001] = {};

	for (int i = 0; i < n; i++) {
		cin >> arr[i];
	}

	int max = -1000000;
	int min = 1000000;
	for (int i = 0; i < n; i++) {
		if (arr[i] > max)	max = arr[i];
		if (arr[i] < min)	min = arr[i];
	}

	cout << min << " " << max << endl;
}
