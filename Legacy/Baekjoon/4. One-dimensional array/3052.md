```c++
// C++
// 1. 나머지를 기록하며 count를 세는 방법
// 1번과 2번 풀이의 메모리와 시간은 같음
// 2번 방법 추천
#include <iostream>

using namespace std;

int main() {
	int arr[11];

	int result[11];
	for (int i = 0; i < 10; i++) {
		result[i] = 42;
	}

	int count = 0;

	for (int i = 0; i < 10; i++) {
		cin >> arr[i];

		for (int j = 0; j < 10; j++) {
			if (arr[i] % 42 != result[j]) {
				if (result[j] == 42) {
					result[j] = arr[i] % 42;
					count++;
					break;
				}
				else {
					continue;
				}
			}
			else {
				break;
			}
		}
	}

	cout << count << endl;
}

// 2. 나올 수 있는 길이 42짜리의 나머지 배열을 만들어 기록하는 방법
#include <iostream>

using namespace std;

int main() {
	int arr[43] = {};

	int a;
	for (int i = 0; i < 10; i++) {
		cin >> a;
		arr[a % 42] = 1;
	}

	int count = 0;
	for (int i = 0; i < 42; i++) {
		if (arr[i] > 0)	count++;
	}

	cout << count << endl;
}
```

```c
// C
// C++의 2번 방법 활용
#include <stdio.h>

int main() {
	int result[43] = {0};

	int a;
	for (int i = 0; i < 10; i++) {
		scanf_s("%d", &a);
		result[a % 42] = 1;
	}

	int count = 0;
	for (int i = 0; i < sizeof(result) / sizeof(int); i++) {
		if (result[i] > 0) {
			count++;
		}
	}

	printf("%d", count);
	return 0;
}
```
